    ti = kwargs['ti']
    table = DATASETS[dataset_key]
    out_path = f'{PARQUET_DIR}/raw_{dataset_key}.parquet'

    # {% endverbatim %}{% trans "--- เริ่มจับเวลา ---" %}{% verbatim %}
    t_start = time.perf_counter()

    # {% endverbatim %}{% trans "เชื่อม SQL Server ด้วย pyodbc" %}{% verbatim %}

    cursor = conn.cursor()

    # {% endverbatim %}{% trans "ดึงข้อมูลจาก Table" %}{% verbatim %}
    cursor.execute(f'SELECT * FROM {table}')
    columns = [desc[0] for desc in cursor.description] # {% endverbatim %}{% trans "ชื่อ Column" %}{% verbatim %}
    rows = cursor.fetchall() # {% endverbatim %}{% trans "ข้อมูลทุกแถว" %}{% verbatim %}
    conn.close()

    # {% endverbatim %}{% trans "แปลงเป็น dict" %}{% verbatim %}
    data = {
        col_name: [row[i] for row in rows]
        for i, col_name in enumerate(columns)
    }

    # {% endverbatim %}{% trans "แปลงเป็น PyArrow Table แล้วบันทึกเป็น Parquet" %}{% verbatim %}
    pq.write_table(pa.table(data), out_path)

    # {% endverbatim %}{% trans "--- หยุดจับเวลา ---" %}{% verbatim %}
    extract_sec = time.perf_counter() - t_start

    # {% endverbatim %}{% trans "ส่งค่าผ่าน XCom ไปให้ Task ถัดไป" %}{% verbatim %}
    ti.xcom_push(key=f'extract_sec_{dataset_key}', value=extract_sec)
    ti.xcom_push(key=f'parquet_raw_{dataset_key}', value=out_path)

    print(f'[Extract] {dataset_key}: {len(rows)} rows | {extract_sec:.3f}s')

# ============================================================================
# TASK 2 — TRANSFORM
# ============================================================================
def transform(dataset_key: str, **kwargs):
    ti = kwargs['ti']
    in_path = ti.xcom_pull(key=f'parquet_raw_{dataset_key}')
    out_path = f'{PARQUET_DIR}/clean_{dataset_key}.parquet'

    t_start = time.perf_counter()

    {% trans "# สร้าง SparkSession" %}
    spark = (SparkSession.builder
        .appName('fraud_etl')
        .config('spark.jars', JDBC_JAR)
        .getOrCreate())

    {% trans "# สร้าง SHA-256 UDF" %}
    hash_udf = udf(
        lambda x: hashlib.sha256(str(x).encode()).hexdigest()
                  if x is not None else None,
        StringType()
    )

    df = spark.read.parquet(in_path)

    df = (df
        {% trans "# Step 1: Hash ข้อมูลส่วนตัว + Rename Columns" %}
        .withColumn('cc_num_hash', hash_udf(col('cc_num')))
        .withColumn('first_hash', hash_udf(col('first')))
        .withColumn('last_hash', hash_udf(col('last')))
        .withColumn('dob_hash', hash_udf(col('dob')))

        {% trans "# Step 2: แปลง DateTime + Rename Columns" %}
        .withColumn('trans_datetime',
                    to_timestamp(col('trans_date_trans_time'),
                                'yyyy-MM-dd HH:mm:ss'))

        {% trans "# Step 3: Filter amt ผิดปกติ" %}
        .filter(col('amt').isNotNull())
        .filter(col('amt') > 0)

        {% trans "# Step 4: Select เฉพาะ column ที่ต้องการ Load" %}
        .select(
            col('trans_datetime'),
            col('cc_num_hash'),
            col('merchant'),
            col('category'),
            col('amt').cast('double'),
            col('first_hash'),
            col('last_hash'),
            col('gender'),
            col('city'),
            col('state'),
            col('city_pop').cast('int'),
            col('job'),
            col('dob_hash'),
            col('trans_num'),
            col('is_fraud').cast('int')
        )
    )

    df.write.mode('overwrite').parquet(out_path)
    spark.stop()

    transform_sec = time.perf_counter() - t_start

    ti.xcom_push(key=f'transform_sec_{dataset_key}', value=transform_sec)
    ti.xcom_push(key=f'parquet_clean_{dataset_key}', value=out_path)

    print(f'[Transform] {dataset_key}: {transform_sec:.3f}s')

def load(dataset_key: str, **kwargs):

    spark = (SparkSession.builder
        .appName('fraud_etl')
        .config('spark.jars', JDBC_JAR)
        .getOrCreate())

    df = spark.read.parquet(in_path)

    (df.write
        .format('jdbc')
        .option('url', MYSQL_URL)
        .option('dbtable', 'fraud_dl.fraud_clean_airflow')
        .option('driver', MYSQL_PROP['driver'])
        .option('user', MYSQL_PROP['user'])
        .option('password', MYSQL_PROP['password'])
        .mode('append')
        .save())

    spark.stop()

    load_sec = time.perf_counter() - t_start

    ti.xcom_push(key=f'load_sec_{dataset_key}', value=load_sec)

    print(f'[Load] {dataset_key}: {load_sec:.3f}s')
