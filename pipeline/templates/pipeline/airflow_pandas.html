{% trans "เปิด VS Code สร้างไฟล์" %} fraud_etl_pandas.py {% trans "ในโฟลเดอร์ dags วางโค้ดนี้" %}:

{% verbatim %}# ============================================================================
# fraud_etl_pandas.py
# Apache Airflow 2.8.0 + pandas ETL Pipeline
# Credit Card Fraud Detection Dataset
# ============================================================================

import os
import csv
import time
import hashlib

import pyodbc
import pandas as pd
import pymysql
from datetime import datetime
from dotenv import load_dotenv

from airflow import DAG
from airflow.operators.python import PythonOperator

# ============================================================================
{% endverbatim %}{% trans "# โหลด .env" %}{% verbatim %}
# ============================================================================
load_dotenv(os.path.join(os.path.dirname(__file__), '.env'))

# ============================================================================
# Config
# ============================================================================
MSSQL_CONN = (
    f"DRIVER={{ODBC Driver 17 for SQL Server}};"
    f"SERVER={os.getenv('MSSQL_SERVER')};"
    f"DATABASE={os.getenv('MSSQL_DATABASE')};"
    f"UID={os.getenv('MSSQL_UID')};"
    f"PWD={os.getenv('MSSQL_PWD')}"
)

PARQUET_DIR = "/tmp/fraud_etl"
CSV_LOG = "/tmp/fraud_etl/timing_log_airflow_pandas.csv"
DATASETS = {"10k": "fraud_10k", "100k": "fraud_100k", "1m": "fraud_1m"}

# ============================================================================
# TASK 1 — EXTRACT
# ============================================================================
def extract(dataset_key: str, **kwargs):
    ti = kwargs['ti']
    table = DATASETS[dataset_key]
    out_path = f'{PARQUET_DIR}/pandas_raw_{dataset_key}.parquet'

    t_start = time.perf_counter()

    conn = pyodbc.connect(MSSQL_CONN)
    df = pd.read_sql(f'SELECT * FROM {table}', conn)
    conn.close()

    df.to_parquet(out_path, index=False)

    extract_sec = time.perf_counter() - t_start

    ti.xcom_push(key=f'extract_sec_{dataset_key}', value=extract_sec)
    ti.xcom_push(key=f'parquet_raw_{dataset_key}', value=out_path)

    print(f'[Extract] {dataset_key}: {len(df)} rows | {extract_sec:.3f}s')

# ============================================================================
# TASK 2 — TRANSFORM
# ============================================================================
def transform(dataset_key: str, **kwargs):
    ti = kwargs['ti']
    in_path = ti.xcom_pull(key=f'parquet_raw_{dataset_key}')
    out_path = f'{PARQUET_DIR}/pandas_clean_{dataset_key}.parquet'

    t_start = time.perf_counter()

    df = pd.read_parquet(in_path)

    {% endverbatim %}{% trans "# Step 1: Hash ข้อมูลส่วนตัว" %}{% verbatim %}
    def sha256(x):
        return hashlib.sha256(str(x).encode()).hexdigest() if pd.notna(x) else None

    df['cc_num_hash'] = df['cc_num'].apply(sha256)
    df['first_hash'] = df['first'].apply(sha256)
    df['last_hash'] = df['last'].apply(sha256)
    df['dob_hash'] = df['dob'].apply(sha256)

    {% endverbatim %}{% trans "# Step 2: แปลง DateTime" %}{% verbatim %}
    df['trans_datetime'] = pd.to_datetime(df['trans_date_trans_time'])

    # Step 3: Filter
    df = df[df['amt'].notna()]
    df = df[df['amt'] > 0]

    {% endverbatim %}{% trans "# Step 4: Select เฉพาะ column ที่ต้องการ" %}{% verbatim %}
    df = df[[
        'trans_datetime', 'cc_num_hash', 'merchant', 'category',
        'amt', 'first_hash', 'last_hash', 'gender', 'city', 'state',
        'city_pop', 'job', 'dob_hash', 'trans_num', 'is_fraud'
    ]]

    df.to_parquet(out_path, index=False)

    transform_sec = time.perf_counter() - t_start

    ti.xcom_push(key=f'transform_sec_{dataset_key}', value=transform_sec)
    ti.xcom_push(key=f'parquet_clean_{dataset_key}', value=out_path)

    print(f'[Transform] {dataset_key}: {transform_sec:.3f}s')

# ============================================================================
# TASK 3 — LOAD
# ============================================================================
def load(dataset_key: str, **kwargs):
    ti = kwargs['ti']
    in_path = ti.xcom_pull(key=f'parquet_clean_{dataset_key}')

    t_start = time.perf_counter()

    df = pd.read_parquet(in_path)

    conn = pymysql.connect(
        host=os.getenv('MYSQL_HOST'),
        port=int(os.getenv('MYSQL_PORT', '3306')),
        user=os.getenv('MYSQL_USER'),
        password=os.getenv('MYSQL_PWD'),
        database='fraud_dl'
    )
    cursor = conn.cursor()

    cols = ','.join(df.columns)
    placeholders = ','.join(['%s'] * len(df.columns))
    sql = f"INSERT INTO fraud_clean_airflow_pandas ({cols}) VALUES ({placeholders})"

    batch_size = 1000
    rows = [tuple(row) for row in df.itertuples(index=False, name=None)]

    for i in range(0, len(rows), batch_size):
        cursor.executemany(sql, rows[i:i+batch_size])

    conn.commit()
    cursor.close()
    conn.close()

    load_sec = time.perf_counter() - t_start

    ti.xcom_push(key=f'load_sec_{dataset_key}', value=load_sec)

    print(f'[Load] {dataset_key}: {load_sec:.3f}s')

# ============================================================================
# TASK 4 — LOG TIMING
# ============================================================================
def log_timing(dataset_key: str, **kwargs):
    ti = kwargs['ti']

    extract_sec = ti.xcom_pull(key=f'extract_sec_{dataset_key}')
    transform_sec = ti.xcom_pull(key=f'transform_sec_{dataset_key}')
    load_sec = ti.xcom_pull(key=f'load_sec_{dataset_key}')

    write_header = not os.path.exists(CSV_LOG)

    with open(CSV_LOG, 'a', newline='') as f:
        writer = csv.writer(f)
        if write_header:
            writer.writerow(['dataset', 'extract_sec', 'transform_sec', 'load_sec'])
        writer.writerow([dataset_key,
                        round(extract_sec, 4),
                        round(transform_sec, 4),
                        round(load_sec, 4)])

    print(f'[Log] {dataset_key}: extract={extract_sec:.3f}s | '
          f'transform={transform_sec:.3f}s | load={load_sec:.3f}s')

# ============================================================================
# DAG Definition
# ============================================================================
default_args = {'owner': 'airflow', 'retries': 0}

with DAG(
    dag_id='fraud_etl_pandas',
    default_args=default_args,
    schedule_interval=None,
    start_date=datetime(2024, 1, 1),
    catchup=False
) as dag:

    for key in ['10k', '100k', '1m']:
        extract_task = PythonOperator(
            task_id=f'extract_{key}',
            python_callable=extract,
            op_kwargs={'dataset_key': key}
        )

        transform_task = PythonOperator(
            task_id=f'transform_{key}',
            python_callable=transform,
            op_kwargs={'dataset_key': key}
        )

        load_task = PythonOperator(
            task_id=f'load_{key}',
            python_callable=load,
            op_kwargs={'dataset_key': key}
        )

        log_timing_task = PythonOperator(
            task_id=f'log_timing_{key}',
            python_callable=log_timing,
            op_kwargs={'dataset_key': key}
        )

        extract_task >> transform_task >> load_task >> log_timing_task{% endverbatim %}

{% trans "สาเหตุที่ Pandas เร็วกว่าเพราะส่งข้อมูลหลายรายการพร้อมกันในครั้งเดียว ทำให้ใช้เวลารวมน้อยกว่ามาก[10] ส่วน PySpark ช้ากว่านั้นยังไม่สามารถระบุได้อย่างชัดเจนมากนัก แต่จากการวิเคราะห์คิดว่าเป็นเพราะวิธีที่มันส่งข้อมูลเข้าฐานข้อมูลนั้นมาจาก JDBC ที่ไม่ได้ตั้งค่าแบบส่งข้อมูลหลายรายการพร้อมกันในครั้งเดียวซึ่งส่งผลให้มีการส่งข้อมูลทีละแถว เมื่อข้อมูลมี 1,000,000 แถว PySpark จะส่งข้อมูลทีละแถวและทุกรอบต้องรอให้ MySQL ตอบรับก่อนจึงจะส่งรอบถัดไปได้ทำให้เวลารวมพุ่งสูงถึง 955 วินาที" %}
